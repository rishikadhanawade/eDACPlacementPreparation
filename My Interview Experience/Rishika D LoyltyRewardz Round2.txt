Write Program for factorial (Recursive & Iterative)


//Iterative
int n=5;
int fact=1;

for(int i=1;i<=n;i++)
{
	fact=fact*i;
}


//Recursive
int fact(int num)
{
	if(num==1||num==0)
		return 1;
	else
		return num*fact(num-1);
}
